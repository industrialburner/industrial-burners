<script>
  function sendInquiry() {
    const btn    = document.getElementById('submitBtn');
    const status = document.getElementById('formStatus');

    // Grab values
    const name    = document.getElementById('from_name').value.trim();
    const email   = document.getElementById('from_email').value.trim();
    const phone   = document.getElementById('phone').value.trim();
    const company = document.getElementById('company').value.trim();
    const product = document.getElementById('product').value;
    const message = document.getElementById('message').value.trim();

    // Basic validation
    if (!name || !email || !phone) {
      showStatus('⚠ Please fill in Name, Email and Phone.', '#C8963E', '#1a1500');
      return;
    }

    // Loading state
    btn.disabled = true;
    btn.textContent = '⏳ Sending…';
    status.style.display = 'none';

    emailjs.send(EMAILJS_SERVICE_ID, EMAILJS_TEMPLATE_ID, {
      from_name : name,
      company   : company || '—',
      phone     : phone,
      from_email: email,
      product   : product || 'Not specified',
      message   : message || '—'
    })
    .then(() => {
      showStatus('✅ Inquiry sent! We will contact you shortly.', '#22c55e', '#0a1a10');
      btn.textContent = '✓ Sent!';
      // Clear fields
      ['from_name','company','phone','from_email','message'].forEach(id => {
        document.getElementById(id).value = '';
      });
      document.getElementById('product').selectedIndex = 0;
      setTimeout(() => { btn.disabled = false; btn.textContent = '⚡ Send Inquiry'; }, 4000);
    })
    .catch(err => {
      console.error('EmailJS error:', err);
      const errMsg = err?.text || err?.message || JSON.stringify(err);
      showStatus('❌ Error: ' + errMsg, '#FF4500', '#1a0800');
      btn.disabled = false;
      btn.textContent = '⚡ Send Inquiry';
    });
  }

  function showStatus(msg, color, bg) {
    const s = document.getElementById('formStatus');
    s.textContent = msg;
    s.style.display = 'block';
    s.style.color = color;
    s.style.background = bg;
    s.style.border = '1px solid ' + color + '44';
  }
</script>
